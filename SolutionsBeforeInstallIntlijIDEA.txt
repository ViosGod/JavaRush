*/

{

    {
      String s = "I don't get excited by money, they soothe me.";
      print5Times(s);//add your code here
    }


    public static void print5Times(String s)
    {
        
        System.out.println(s);

        System.out.println(s);
        System.out.println(s);
        System.out.println(s);
        System.out.println(s);
        System.out.println(s);
        System.out.println(s);
        System.out.println(s);
    }
}
